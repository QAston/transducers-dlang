Module providing functions for working with ranges.

auto  transduceSource(R, Transducer)(R inputRange, Transducer t) if (isInputRange!R && isTransducer!(Transducer, ElementType!R));

Returns a lazy range, each item is taken from given inputRange and lazily transformed by provided transducer t.

Examples
import transduced.transducers;

auto res = transduceSource([1, 2, 3, 4, 5], chunkMapper!((scope int[] view) => view.dup)(3)).array();
assert(res == [[1, 2, 3], [4, 5]]);
Examples
import transduced.transducers;

auto res = transduceSource([1, 2, 3, 4], comp(taker(2), mapper!minus, mapper!twice));
assert(!res.empty);
assert(res.front == -2);
res.popFront();
assert(!res.empty);
assert(res.front == -4);
res.popFront();
assert(res.empty);

auto  into(R, Transducer, Out)(R from, Transducer t, Out to) if (isInputRange!R && isTransducer!(Transducer, ElementType!R));

Populates output range to with contents of input range from transformed by transducer t.

Examples
import std.array;
import transduced.transducers;

auto output = appender!(int[])();

[1, 2, 3, 4].into(comp(taker(2), mapper!minus, mapper!twice), output);
assert(output.data == [-2, -4]);

auto  transduceSink(InputElementType, Transducer, OutputRange)(Transducer t, OutputRange o) if (isTransducer!(Transducer, InputElementType) && isOutputRange!(OutputRange, Transducer.OutputType!InputElementType));

Returns an ExtendedOutputRange (see transduced.core.isExtendedOutputRange) of type TransducedSink which forwards input transformed by transducer t to OutputRange o. o has to take OutElementType as input.

Examples
import std.array;
import transduced.transducers;

auto output = appender!(int[][])();
auto transducedOutput = transduceSink!int(chunkMapper!((scope int[] view) => view.dup)(3),
    output);
put(transducedOutput, 1);
assert(output.data == []);
put(transducedOutput, 2);
put(transducedOutput, 3);
assert(output.data == [[1, 2, 3]]);
put(transducedOutput, 4);
put(transducedOutput, 5);
assert(output.data == [[1, 2, 3]]);
// flushing is optional, skipping it will just ignore the buffered data on destruction, can also flush multiple times.
transducedOutput.flush();
assert(output.data == [[1, 2, 3], [4, 5]]);
Examples
import std.array;
import transduced.transducers;

auto output = appender!(int[])();
auto transducedOutput = transduceSink!int(mapper!((x) => -x), output);
static assert(isOutputRange!(typeof(transducedOutput), int));
static assert(isExtendedOutputRange!(typeof(transducedOutput), int));
put(transducedOutput, 1);
assert(output.data == [-1]);
put(transducedOutput, 2);
assert(output.data == [-1, -2]);
put(transducedOutput, 3);
assert(output.data == [-1, -2, -3]);
assert(!transducedOutput.isDone());
Examples
import std.array;
import transduced.transducers;

auto output = appender!(int[])();
auto transducedOutput = transduceSink!int(taker(2), output);
put(transducedOutput, 1);
assert(!transducedOutput.isDone());
assert(output.data == [1]);
put(transducedOutput, 2);
assert(output.data == [1, 2]);
put(transducedOutput, 3);
assert(output.data == [1, 2]);
assert(transducedOutput.isDone());
